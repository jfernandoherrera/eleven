CREATE DATABASE IF NOT EXISTS eleven;
USE eleven;
CREATE TABLE IF NOT EXISTS user (id int(11) NOT NULL AUTO_INCREMENT, name varchar(100) NOT NULL, pass varchar(5000) NOT NULL, PRIMARY KEY (id));
CREATE TABLE IF NOT EXISTS player (id int(11) NOT NULL AUTO_INCREMENT, number varchar(100) NOT NULL, PRIMARY KEY (id));
CREATE TABLE IF NOT EXISTS team (id int(11) NOT NULL AUTO_INCREMENT, name varchar(100) NOT NULL, PRIMARY KEY (id));
CREATE TABLE IF NOT EXISTS playerTeam (id int(11) NOT NULL AUTO_INCREMENT, idPlayer varchar(100) NOT NULL, idTeam varchar(5000) NOT NULL, PRIMARY KEY (id));
